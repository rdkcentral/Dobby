97 #define __AI_LOG_PRINTF(level, fmt, ...) \
98  do { \
99  if (__builtin_expect(((level) <= __ai_debug_log_level),0)) \
100  __ai_debug_log_printf((level), __FILE__, __FUNCTION__, __LINE__, fmt, ##__VA_ARGS__); \
