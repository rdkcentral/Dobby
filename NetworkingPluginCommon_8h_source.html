36// creates an in_addr_t type from ip address
37#define INADDR_CREATE(a, b, c, d) \
38 ( ((((in_addr_t)(a)) << 24) & 0xff000000) | \
39 ((((in_addr_t)(b)) << 16) & 0x00ff0000) | \
40 ((((in_addr_t)(c)) << 8) & 0x0000ff00) | \
41 ((((in_addr_t)(d)) << 0) & 0x000000ff) )
